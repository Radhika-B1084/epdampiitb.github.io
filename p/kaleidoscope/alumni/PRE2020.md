      <!-- JS filtering -->
      <script>
        const searchBox = document.getElementById("searchBox");
        const fieldFilter = document.getElementById("fieldFilter");
        const destinationFilter = document.getElementById("destinationFilter");
        const placeFilter = document.getElementById("placeFilter");
        const entries = document.querySelectorAll(".alumni-entry");

        function filterEntries() {
          const searchTerm = searchBox.value.toLowerCase();
          const fieldTerm = fieldFilter.value.toLowerCase();
          const destTerm = destinationFilter.value.toLowerCase();
          const placeTerm = placeFilter.value.toLowerCase();

          entries.forEach(entry => {
	    const text = entry.getAttribute("data-text");
            const field = entry.dataset.field;
            const dest = entry.dataset.destination;
            const place = entry.dataset.place;

            const matchesSearch = text.includes(searchTerm);
            const matchesField = !fieldTerm || field === fieldTerm;
            const matchesDest = !destTerm || dest === destTerm;
            const matchesPlace = !placeTerm || place === placeTerm;

            entry.style.display = (matchesSearch && matchesField && matchesDest && matchesPlace) ? "block" : "none";
          });
        }

        searchBox.addEventListener("input", filterEntries);
        fieldFilter.addEventListener("change", filterEntries);
        destinationFilter.addEventListener("change", filterEntries);
        placeFilter.addEventListener("change", filterEntries);

        // Activate Select2
        document.addEventListener("DOMContentLoaded", function () {
          $('.select2').select2();
        });
      </script>
